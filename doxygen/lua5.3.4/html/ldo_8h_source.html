14 /*
15 ** Macro to check stack size and grow stack if needed. Parameters
16 ** 'pre'/'pos' allow the macro to preserve a pointer into the
17 ** stack across reallocations, doing the work only when needed.
18 ** 'condmovestack' is used in heavy tests to force a stack reallocation
19 ** at every check.
20 */
21 #define luaD_checkstackaux(L, n, pre, pos) \
22  if (L->stack_last - L->top <= (n)) \
23  { \
24  pre; \
25  luaD_growstack(L, n); \
26  pos; \
27  } \
28  else \
29  { \
30  condmovestack(L, pre, pos); \
31  }
32 
33 /* In general, 'pre'/'pos' are empty (nothing to save) */
34 #define luaD_checkstack(L, n) luaD_checkstackaux(L, n, (void)0, (void)0)

36 #define savestack(L, p) ((char*)(p) - (char*)L->stack)
37 #define restorestack(L, n) ((TValue*)((char*)L->stack + (n)))
