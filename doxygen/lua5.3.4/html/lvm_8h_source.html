43 /*
44 ** fast track for 'gettable': if 't' is a table and 't[k]' is not nil,
45 ** return 1 with 'slot' pointing to 't[k]' (final result). Otherwise,
46 ** return 0 (meaning it will have to check metamethod) with 'slot'
47 ** pointing to a nil 't[k]' (if 't' is a table) or NULL (otherwise).
48 ** 'f' is the raw get function to use.
49 */
50 #define luaV_fastget(L, t, k, slot, f) \
51  (!ttistable(t) ? (slot = NULL, 0) /* not a table; 'slot' is NULL and result is 0 */ \
52  : (slot = f(hvalue(t), k), /* else, do raw access */ \
53  !ttisnil(slot))) /* result not nil? */
54 
55 /*
56 ** standard implementation for 'gettable'
57 */
58 #define luaV_gettable(L, t, k, v) \
59  { \
60  const TValue* slot; \
61  if (luaV_fastget(L, t, k, slot, luaH_get)) \
62  { \
63  setobj2s(L, v, slot); \
64  } \
65  else \
66  luaV_finishget(L, t, k, v, slot); \
67  }
68 
69 /*
70 ** Fast track for set table. If 't' is a table and 't[k]' is not nil,
71 ** call GC barrier, do a raw 't[k]=v', and return true; otherwise,
72 ** return false with 'slot' equal to NULL (if 't' is not a table) or
73 ** 'nil'. (This is needed by 'luaV_finishget'.) Note that, if the macro
74 ** returns true, there is no need to 'invalidateTMcache', because the
75 ** call is not creating a new entry.
76 */
77 #define luaV_fastset(L, t, k, slot, f, v) \
78  (!ttistable(t) \
79  ? (slot = NULL, 0) \
80  : (slot = f(hvalue(t), k), \
81  ttisnil(slot) ? 0 : (luaC_barrierback(L, hvalue(t), v), setobj2t(L, cast(TValue*, slot), v), 1)))
82 
83 #define luaV_settable(L, t, k, v) \
84  { \
85  const TValue* slot; \
86  if (!luaV_fastset(L, t, k, slot, luaH_get, v)) \
87  luaV_finishset(L, t, k, v, slot); \
88  }
