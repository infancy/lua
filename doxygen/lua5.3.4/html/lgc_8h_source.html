13 /*
14 ** Collectable objects may have one of three colors: white, which
15 ** means the object is not marked; gray, which means the
16 ** object is marked, but its references may be not marked; and
17 ** black, which means that the object and all its references are marked.
18 ** The main invariant of the garbage collector, while marking objects,
19 ** is that a black object can never point to a white one. Moreover,
20 ** any gray object must be in a "gray list" (gray, grayagain, weak,
21 ** allweak, ephemeron) so that it can be visited again before finishing
22 ** the collection cycle. These lists have no meaning when the invariant
23 ** is not being enforced (e.g., sweep phase).
24 */
25

27 #if !defined(GCSTEPSIZE)
28  /* ~100 small strings */

30 #endif

46 /*
47 ** macro to tell when main invariant (white objects cannot point to black
48 ** ones) must be kept. During a collection, the sweep
49 ** phase may break the invariant, as objects turned white may point to
50 ** still-black objects. The invariant is restored when sweep ends and
51 ** all objects are white again.
52 */
53 
54 #define keepinvariant(g) ((g)->gcstate <= GCSatomic)

56 /*
57 ** some useful bit tricks
58 */
59 #define resetbits(x, m) ((x) &= cast(lu_byte, ~(m)))
60 #define setbits(x, m) ((x) |= (m))
61 #define testbits(x, m) ((x) & (m))
62 #define bitmask(b) (1 << (b))
63 #define bit2mask(b1, b2) (bitmask(b1) | bitmask(b2))
64 #define l_setbit(x, b) setbits(x, bitmask(b))
65 #define resetbit(x, b) resetbits(x, bitmask(b))
66 #define testbit(x, b) testbits(x, bitmask(b))
67 
68 /* Layout for bit use in 'marked' field: */
69 #define WHITE0BIT 0 /* object is white (type 0) */
70 #define WHITE1BIT 1 /* object is white (type 1) */
71 #define BLACKBIT 2 /* object is black */
72 #define FINALIZEDBIT 3 /* object has been marked for finalization */
73 /* bit 7 is currently used by tests (luaL_checkmemory) */
74 
75 #define WHITEBITS bit2mask(WHITE0BIT, WHITE1BIT)
76 
77 #define iswhite(x) testbits((x)->marked, WHITEBITS)
78 #define isblack(x) testbit((x)->marked, BLACKBIT)
79 #define isgray(x) /* neither white nor black */ (!testbits((x)->marked, WHITEBITS | bitmask(BLACKBIT)))
80 
81 #define tofinalize(x) testbit((x)->marked, FINALIZEDBIT)
82 
83 #define otherwhite(g) ((g)->currentwhite ^ WHITEBITS)
84 #define isdeadm(ow, m) (!(((m) ^ WHITEBITS) & (ow)))
85 #define isdead(g, v) isdeadm(otherwhite(g), (v)->marked)

87 #define changewhite(x) ((x)->marked ^= WHITEBITS)

92 /*
93 ** Does one step of collection when debt becomes positive. 'pre'/'pos'
94 ** allows some adjustments to be done only when needed. macro
95 ** 'condchangemem' is used only for heavy tests (forcing a full
96 ** GC cycle on every opportunity)
97 */
98 #define luaC_condGC(L, pre, pos) \
99  { \
100  if (G(L)->GCdebt > 0) \
101  { \
102  pre; \
103  luaC_step(L); \
104  pos; \
105  }; \
106  condchangemem(L, pre, pos); \
107  }

109 /* more often than not, 'pre'/'pos' are empty */
110 #define luaC_checkGC(L) luaC_condGC(L, (void)0, (void)0)
