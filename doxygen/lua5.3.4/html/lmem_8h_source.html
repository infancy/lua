15 /*
16 ** This macro reallocs a vector 'b' from 'on' to 'n' elements, where
17 ** each element has size 'e'. In case of arithmetic overflow of the
18 ** product 'n'*'e', it raises an error (calling 'luaM_toobig'). Because
19 ** 'e' is always constant, it avoids the runtime division MAX_SIZET/(e).
20 **
21 ** (The macro is somewhat complex to avoid warnings: The 'sizeof'
22 ** comparison avoids a runtime comparison when overflow cannot occur.
23 ** The compiler should be able to optimize the real test by itself, but
24 ** when it does it, it may give a warning about "comparison is always
25 ** false due to limited range of data type"; the +1 tricks the compiler,
26 ** avoiding this warning but also this optimization.)
27 */
28 #define luaM_reallocv(L, b, on, n, e) \
29  (((sizeof(n) >= sizeof(size_t) && cast(size_t, (n)) + 1 > MAX_SIZET / (e)) ? luaM_toobig(L) : cast_void(0)), \
30  luaM_realloc_(L, (b), (on) * (e), (n) * (e)))
31 
32 /*
33 ** Arrays of chars do not need any test
34 */
35 #define luaM_reallocvchar(L, b, on, n) cast(char*, luaM_realloc_(L, (b), (on) * sizeof(char), (n) * sizeof(char)))

39 #define luaM_freearray(L, b, n) luaM_realloc_(L, (b), (n) * sizeof(*(b)), 0)
